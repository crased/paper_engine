# Paper Engine v1.0 - Python Dependencies

# ============================================================
# CORE DEPENDENCIES ONLY
# ============================================================
# This file contains only the core dependencies that install quickly.
# The script will prompt you to install LLM providers interactively.
# Heavy packages (label-studio, ultralytics) should be installed manually.

# Keyboard and mouse control
pynput>=1.7.6

# Environment variable management
python-dotenv>=1.0.0

# YAML parsing for config
PyYAML>=6.0

# ============================================================
# INSTALLATION INSTRUCTIONS
# ============================================================
#
# 1. Install core dependencies (fast):
#    pip install -r information/requirements.txt
#
# 2. Run Paper Engine - it will prompt you to install an LLM provider:
#    ./paperengine
#    (The script will detect missing LLM and offer installation)
#
# 3. Manually install heavy packages when needed:
#    pip install label-studio    # For annotation (Step 2)
#    pip install ultralytics      # For YOLO training (Step 3)
#
# ============================================================
# LLM PROVIDERS (installed by script when you choose one)
# ============================================================
#
# google-generativeai>=0.3.0  # Free tier (recommended)
# anthropic>=0.18.0           # Claude (paid)
# openai>=1.12.0              # GPT (paid)
#
# The script will ask which provider you want and install it automatically.
#
# ============================================================
# SYSTEM REQUIREMENTS
# ============================================================
#
# Linux:
#   - wine: sudo apt install wine (for Windows game execution)
#   - screenshot tool (choose one):
#       - flameshot (recommended): sudo apt install flameshot
#       - scrot (alternative): sudo apt install scrot
#       - imagemagick (alternative): sudo apt install imagemagick
#
# Windows:
#   - No additional system tools needed
